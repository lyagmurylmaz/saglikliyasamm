    otherItem.classList.remove('active');

    // Testimonial slider
    const testimonialContainer = document.querySelector('.testimonial-container');
    const sliderDots = document.querySelectorAll('.slider-dot');
    const testimonials = document.querySelectorAll('.testimonial');
    sliderDots.forEach((dot, index) => {
    dot.addEventListener('click', () => {
    // Update active dot
    sliderDots.forEach(d => d.classList.remove('active'));
    dot.classList.add('active');
    // Slide to the selected testimonial
    testimonialContainer.scrollLeft = index * testimonialContainer.clientWidth;
    });
    });
    // Auto slide testimonials every 5 seconds
    let currentSlide = 0;
    function autoSlide() {
    currentSlide = (currentSlide + 1) % testimonials.length;
    // Update active dot
    sliderDots.forEach(d => d.classList.remove('active'));
    sliderDots[currentSlide].classList.add('active');
    // Slide to the next testimonial
    testimonialContainer.scrollLeft = currentSlide * testimonialContainer.clientWidth;
    }
    let slideInterval = setInterval(autoSlide, 5000);
    // Pause auto slide when user interacts with testimonials
    testimonialContainer.addEventListener('mouseenter', () => {
    clearInterval(slideInterval);
    });
    testimonialContainer.addEventListener('mouseleave', () => {
    slideInterval = setInterval(autoSlide, 5000);
    });
    // Mobile menu for dropdown
    const dropdowns = document.querySelectorAll('.dropdown');
    if (window.innerWidth <= 768) {
    dropdowns.forEach(dropdown => {
    dropdown.addEventListener('click', (e) => {
    e.preventDefault();
